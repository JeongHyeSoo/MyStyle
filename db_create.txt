create table post_Information(

POST_NUM INT PRIMARY KEY DEFAULT '0',
USER_NUM INT NOT NULL,
TIME TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
TITLE VARCHAR(100) NOT NULL,
STYLE_IMG MEDIUMBLOB NOT NULL,
COMMENT VARCHAR(20000),
WARNING_CNT INT DEFAULT '0',
HEART_CNT INT DEFAULT '0'

);

create table post_product_infotmation(

POST_NUM INT PRIMARY KEY,
PRODUCT_NUM INT NOT NULL,
PRODUCT_NAME VARCHAR(100) NOT NULL,
PRODUCT_LINK VARCHAR(500) NOT NULL,
SUB_HEART INT DEFAULT '0'

);


create table post_sub_comment(

POST_NUM INT PRIMARY KEY,
SUB_COMMENT_NUM INT PRIMARY KEY,
SUB_COMMENT VARCHAR(20000) NOT NULL,
SUB_COMMENT_TIME TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP

);
